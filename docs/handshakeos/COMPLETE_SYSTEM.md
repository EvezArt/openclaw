### Usage Examples

**1. Entropy Calculation:**
```typescript
import { calculateEntropy, calculateInformationGain } from './handshakeos/entropy.js';

const probabilities = [0.3, 0.5, 0.2];
const entropy = calculateEntropy(probabilities); // 1.485 bits

const prior = [0.5, 0.5];
const posterior = [0.9, 0.1];
const gain = calculateInformationGain(prior, posterior); // 0.469 bits
```

**2. User Identity Modeling:**
```typescript
import { UserIdentityModel } from './handshakeos/user-identity.js';

const model = new UserIdentityModel('user123');

// Record observations
model.recordInteraction({ action: 'query', timestamp: Date.now() });
model.recordPreference('morning_work', 0.9);

// Predict intent
const intent = model.predictIntent({ context: 'work' });
// { intent: 'productivity', confidence: 0.85 }
```

**3. Network Topology Prediction:**
```typescript
import { NetworkTopologyPredictor } from './handshakeos/network-topology.js';

const predictor = new NetworkTopologyPredictor();

// Observe connections

// Predict without probing
const pred = predictor.predictConnection('node1', 'node2');
// { exists: true, probability: 0.87, confidence: 0.72 }
```
